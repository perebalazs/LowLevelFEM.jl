161 @eval function load_state_acquire(t)
162 # TODO: Replace this by proper atomic operations when available
163 @GC.preserve t llvmcall($("""
164 %ptr = inttoptr i$(Sys.WORD_SIZE) %0 to i8*
165 %rv = load atomic i8, i8* %ptr acquire, align 8
166 ret i8 %rv
167 """), UInt8, Tuple{Ptr{UInt8}},
168 Ptr{UInt8}(pointer_from_objref(t) + fieldoffset(Task, _state_index)))
169 end

171 @inline function getproperty(t::Task, field::Symbol)
172 if field === :state
173 # TODO: this field name should be deprecated in 2.0
174 st = load_state_acquire(t)
175 if st === task_state_runnable
176 return :runnable
177 elseif st === task_state_done
178 return :done
179 elseif st === task_state_failed
180 return :failed
181 else
182 @assert false
183 end
184 elseif field === :backtrace
185 # TODO: this field name should be deprecated in 2.0
186 return current_exceptions(t)[end][2]
187 elseif field === :exception
188 # TODO: this field name should be deprecated in 2.0
189 return t._isexception ? t.result : nothing
190 elseif field === :scope
191 error("Querying `scope` is disallowed. Use `current_scope` instead.")
192 else
193 return getfield(t, field)
194 end
195 end

420 function sync_end(c::Channel{Any})
421 local c_ex
422 while isready(c)
423 r = take!(c)
424 if isa(r, Task)
425 _wait(r)
426 if istaskfailed(r)
427 if !@isdefined(c_ex)
428 c_ex = CompositeException()
429 end
430 push!(c_ex, TaskFailedException(r))
431 end
432 else
433 try
434 wait(r)
435 catch e
436 if !@isdefined(c_ex)
437 c_ex = CompositeException()
438 end
439 push!(c_ex, e)
440 end
441 end
442 end
443 close(c)
444
445 # Capture all waitable objects scheduled after the end of `@sync` and
446 # include them in the exception. This way, the user can check what was
447 # scheduled by examining at the exception object.
448 if isready(c)
449 local racy
450 for r in c
451 if !@isdefined(racy)
452 racy = []
453 end
454 push!(racy, r)
455 end
456 if @isdefined(racy)
457 if !@isdefined(c_ex)
458 c_ex = CompositeException()
459 end
460 # Since this is a clear programming error, show this exception first:
461 pushfirst!(c_ex, ScheduledAfterSyncException(racy))
462 end
463 end
464
465 if @isdefined(c_ex)
466 throw(c_ex)
467 end
468 nothing
469 end

507 """
508 @async
509
510 Wrap an expression in a [`Task`](@ref) and add it to the local machine's scheduler queue.
511
512 Values can be interpolated into `@async` via `\$`, which copies the value directly into the
513 constructed underlying closure. This allows you to insert the _value_ of a variable,
514 isolating the asynchronous code from changes to the variable's value in the current task.
515
516 !!! warning
517 It is strongly encouraged to favor `Threads.@spawn` over `@async` always **even when no
518 parallelism is required** especially in publicly distributed libraries. This is
519 because a use of `@async` disables the migration of the *parent* task across worker
520 threads in the current implementation of Julia. Thus, seemingly innocent use of
521 `@async` in a library function can have a large impact on the performance of very
522 different parts of user applications.
523
524 !!! compat "Julia 1.4"
525 Interpolating values via `\$` is available as of Julia 1.4.
526 """
527 macro async(expr)
528 do_async_macro(expr, __source__)
529 end

626 function _lift_one_interp!(e)
627 letargs = Any[] # store the new gensymed arguments
628 _lift_one_interp_helper(e, false, letargs) # Start out _not_ in a quote context (false)
629 letargs

631 _lift_one_interp_helper(v, _, _) = v
632 function _lift_one_interp_helper(expr::Expr, in_quote_context, letargs)
633 if expr.head === :$
634 if in_quote_context # This $ is simply interpolating out of the quote
635 # Now, we're out of the quote, so any _further_ $ is ours.
636 in_quote_context = false
637 else
638 newarg = gensym()
639 push!(letargs, :($(esc(newarg)) = $(esc(expr.args[1]))))
640 return newarg # Don't recurse into the lifted $() exprs
641 end
642 elseif expr.head === :quote
643 in_quote_context = true # Don't try to lift $ directly out of quotes
644 elseif expr.head === :macrocall
645 return expr # Don't recur into macro calls, since some other macros use $
646 end
647 for (i,e) in enumerate(expr.args)
648 expr.args[i] = _lift_one_interp_helper(e, in_quote_context, letargs)
649 end
650 expr
651 end

791 function enq_work(t::Task)
792 (t._state === task_state_runnable && t.queue === nothing) || error("schedule: Task not runnable")
793
794 # Sticky tasks go into their thread's work queue.
795 if t.sticky
796 tid = Threads.threadid(t)
797 if tid == 0
798 # The task is not yet stuck to a thread. Stick it to the current
799 # thread and do the same to the parent task (the current task) so
800 # that the tasks are correctly co-scheduled (issue #41324).
801 # XXX: Ideally we would be able to unset this.
802 if GC.in_finalizer()
803 # The task was launched in a finalizer. There is no thread to sticky it
804 # to, so just allow it to run anywhere as if it had been non-sticky.
805 t.sticky = false
806 @goto not_sticky
807 else
808 tid = Threads.threadid()
809 ccall(:jl_set_task_tid, Cint, (Any, Cint), t, tid-1)
810 current_task().sticky = true
811 end
812 end
813 push!(workqueue_for(tid), t)
814 else
815 @label not_sticky
816 tp = Threads.threadpool(t)
817 if tp === :foreign || Threads.threadpoolsize(tp) == 1
818 # There's only one thread in the task's assigned thread pool;
819 # use its work queue.
820 tid = (tp === :interactive) ? 1 : Threads.threadpoolsize(:interactive)+1
821 ccall(:jl_set_task_tid, Cint, (Any, Cint), t, tid-1)
822 push!(workqueue_for(tid), t)
823 else
824 # Otherwise, put the task in the multiqueue.
825 Partr.multiq_insert(t, t.priority)
826 tid = 0
827 end
828 end
829 ccall(:jl_wakeup_thread, Cvoid, (Int16,), (tid - 1) % Int16)
830 return t
831 end

835 """
836 schedule(t::Task, [val]; error=false)
837
838 Add a [`Task`](@ref) to the scheduler's queue. This causes the task to run constantly when the system
839 is otherwise idle, unless the task performs a blocking operation such as [`wait`](@ref).
840
841 If a second argument `val` is provided, it will be passed to the task (via the return value of
842 [`yieldto`](@ref)) when it runs again. If `error` is `true`, the value is raised as an exception in
843 the woken task.
844
845 !!! warning
846 It is incorrect to use `schedule` on an arbitrary `Task` that has already been started.
847 See [the API reference](@ref low-level-schedule-wait) for more information.
848
849 !!! warning
850 By default tasks will have the sticky bit set to true `t.sticky`. This models the
851 historic default for [`@async`](@ref). Sticky tasks can only be run on the worker thread
852 they are first scheduled on, and when scheduled will make the task that they were scheduled
853 from sticky. To obtain the behavior of [`Threads.@spawn`](@ref) set the sticky
854 bit manually to `false`.
855
856 # Examples
857 ```jldoctest
858 julia> a5() = sum(i for i in 1:1000);
859
860 julia> b = Task(a5);
861
862 julia> istaskstarted(b)
863 false
864
865 julia> schedule(b);
866
867 julia> yield();
868
869 julia> istaskstarted(b)
870 true
871
872 julia> istaskdone(b)
873 true
874 ```
875 """
876 function schedule(t::Task, @nospecialize(arg); error=false)
877 # schedule a task to be (re)started with the given value or exception
878 t._state === task_state_runnable || Base.error("schedule: Task not runnable")
879 if error
880 t.queue === nothing || Base.list_deletefirst!(t.queue::IntrusiveLinkedList{Task}, t)
881 setfield!(t, :result, arg)
882 setfield!(t, :_isexception, true)
883 else
884 t.queue === nothing || Base.error("schedule: Task not runnable")
885 setfield!(t, :result, arg)
886 end
887 enq_work(t)
888 return t
889 end
